<strong>Exercise 4.6:</strong> Write an expression to determine whether an int value is even or odd.
{% highlight cpp %}
#include <iostream>

int main() {
    constexpr size_t N = 10;
    
    for (size_t i = 0; i != N; i++) {
        if (i % 2) {
            cout << "odd" << endl;
        } else {
            cout << "even" << endl;
        }
    } return 0;
}
{% endhighlight %}
