<strong>Homework 4:</storng> Position

{% highlight java %}
public class Position {
  public static void main(String[] args) {
    double s0 = 10.5;
    double v0 = -12.5;
    double a = 9.8;
    double s = s0;
    
    System.out.println("For s0 = " + s0 + " v0 = " + v0 + ", and a = " + a);
    
    for (int t = 0; t <= 10; t++) {
      System.out.println("At time t = " + t + ", the new position s is: " + (s + v0 * t + 0.5 * a * t * t));
    }
  }
}
{% endhighlight %}


<strong>Homework 4:</storng> Sequence

{% highlight java %}
public class Sequence {
  public static void main(String[] args) {
    printTimes();
    System.out.println();
    printSquares();
    System.out.println();
    printCubes();
  }
  
  public static void printTimes() {
    for (int i = 0; i <= 7; i++)
      System.out.println(i * 2);
  }
  
  public static void printSquares() {
    for (int i = 0; i <= 7; i++) 
      System.out.println(i * i);
  }
  
  public static void printCubes() {
    for (int i = 0; i <= 7; i++) 
      System.out.println(i * i * i);
  }
}
{% endhighlight %}
